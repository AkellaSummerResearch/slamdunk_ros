cmake_minimum_required(VERSION 2.8.12)
project(slamdunk_msgs)

find_package(catkin REQUIRED COMPONENTS
  message_generation
  std_msgs
)

  DIRECTORY msg
  FILES
  BoolStamped.msg

  QualityStamped.msg

  VideoSources.msg
)

  DIRECTORY srv
  FILES Save.srv
)

catkin_package(CATKIN_DEPENDS message_runtime std_msgs)
